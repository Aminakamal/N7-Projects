# Exercice 2 - Affectation avec préférences
# ============================================

# Paramètres
param N;  # Nombre de personnes et de tâches
set PERSONNES := 1..N;
set TACHES := 1..N;
param preference{PERSONNES, TACHES};  # Scores de préférence c(i,j)

# Variables de décision
# x[i,j] = 1 si la personne i est affectée à la tâche j, 0 sinon
var x{PERSONNES, TACHES}, binary;

# Fonction objectif: maximiser la somme des scores de préférence
maximize score_total: sum{i in PERSONNES, j in TACHES} preference[i,j] * x[i,j];

# Contraintes
# Chaque personne effectue exactement une tâche
s.t. une_tache_par_personne{i in PERSONNES}:
    sum{j in TACHES} x[i,j] = 1;

# Chaque tâche est affectée à exactement une personne
s.t. une_personne_par_tache{j in TACHES}:
    sum{i in PERSONNES} x[i,j] = 1;

end;
