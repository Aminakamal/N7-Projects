# ============================================
# Exercice 3 - Cas 1.2 : Avec coûts d'expédition
# ============================================

# Ensembles
set DEMANDES;
set FLUIDES;
set MAGASINS;

# Paramètres
param demande{DEMANDES, FLUIDES} >= 0;
param stock{MAGASINS, FLUIDES} >= 0;
param cout_unitaire{MAGASINS, FLUIDES} >= 0;
param cout_fixe{MAGASINS, DEMANDES} >= 0;      # Coût fixe d'expédition
param cout_variable{MAGASINS, DEMANDES} >= 0;  # Coût variable d'expédition
param M_grand := 10000;  # Grand M pour contraintes d'activation

# Variables de décision
var x{MAGASINS, DEMANDES, FLUIDES} >= 0;  # Quantité fournie
var y{MAGASINS, DEMANDES}, binary;        # 1 si magasin m expédie à demande d
var q{MAGASINS, DEMANDES} >= 0;           # Quantité totale expédiée

# Fonction objectif
minimize cout_total:
    sum{m in MAGASINS, d in DEMANDES, f in FLUIDES} cout_unitaire[m,f] * x[m,d,f]
    + sum{m in MAGASINS, d in DEMANDES} (cout_fixe[m,d] * y[m,d] + cout_variable[m,d] * q[m,d]);

# Contraintes
# Satisfaction des demandes
s.t. satisfaction{d in DEMANDES, f in FLUIDES}:
    sum{m in MAGASINS} x[m,d,f] = demande[d,f];

# Limites de stock
s.t. limite_stock{m in MAGASINS, f in FLUIDES}:
    sum{d in DEMANDES} x[m,d,f] <= stock[m,f];

# Calcul de la quantité totale expédiée
s.t. calcul_quantite{m in MAGASINS, d in DEMANDES}:
    q[m,d] = sum{f in FLUIDES} x[m,d,f];

# Activation du coût fixe (si q > 0 alors y = 1)
s.t. activation_cout_fixe{m in MAGASINS, d in DEMANDES}:
    q[m,d] <= M_grand * y[m,d];

end;
