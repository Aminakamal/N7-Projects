# Exercice 3 - Cas 1.1 : Affectation commandes/magasins
# ============================================

# Ensembles
set DEMANDES;   # Ensemble des commandes/demandes
set FLUIDES;    # Ensemble des fluides/produits
set MAGASINS;   # Ensemble des magasins

# Paramètres
param demande{DEMANDES, FLUIDES} >= 0;     # Quantité demandée de chaque fluide par demande
param stock{MAGASINS, FLUIDES} >= 0;       # Stock disponible de chaque fluide par magasin
param cout{MAGASINS, FLUIDES} >= 0;        # Coût unitaire par magasin et fluide

# Variables de décision
# x[m,d,f] = quantité de fluide f fournie par magasin m à demande d
var x{MAGASINS, DEMANDES, FLUIDES} >= 0;

# Fonction objectif: minimiser le coût total
minimize cout_total: 
    sum{m in MAGASINS, d in DEMANDES, f in FLUIDES} cout[m,f] * x[m,d,f];

# Contraintes
# Satisfaction complète de chaque demande
s.t. satisfaction_demande{d in DEMANDES, f in FLUIDES}:
    sum{m in MAGASINS} x[m,d,f] = demande[d,f];

# Respect des limites de stock par magasin
s.t. limite_stock{m in MAGASINS, f in FLUIDES}:
    sum{d in DEMANDES} x[m,d,f] <= stock[m,f];

# Pour variables entières (articles e-commerce), 
# décommentez les lignes suivantes et commentez la déclaration var ci-dessus
# var x{MAGASINS, DEMANDES, FLUIDES} >= 0, integer;

end;
