# Exercice 3 - Cas 2 : Voyageur de commerce
# ============================================

# Ensembles
set NOEUDS;                           # Tous les noeuds (ALPHA + clients)
param depot symbolic;                 # Le dépôt (ALPHA)
set CLIENTS := NOEUDS diff {depot};   # Les clients uniquement

# Paramètres
param distance{NOEUDS, NOEUDS} >= 0;  # Matrice des distances

# Variables de décision
var x{NOEUDS, NOEUDS}, binary;  # x[i,j] = 1 si on va de i vers j
var u{CLIENTS} >= 1;             # Position dans la tournée (contraintes MTZ)

# Fonction objectif: minimiser la distance totale
minimize distance_totale:
    sum{i in NOEUDS, j in NOEUDS: i <> j} distance[i,j] * x[i,j];

# Contraintes
# Partir du dépôt exactement une fois
s.t. depart_depot:
    sum{j in CLIENTS} x[depot,j] = 1;

# Revenir au dépôt exactement une fois
s.t. retour_depot:
    sum{i in CLIENTS} x[i,depot] = 1;

# Chaque client est visité exactement une fois (arrivée)
s.t. visite_client_entree{j in CLIENTS}:
    sum{i in NOEUDS: i <> j} x[i,j] = 1;

# Chaque client est quitté exactement une fois (sortie)
s.t. visite_client_sortie{i in CLIENTS}:
    sum{j in NOEUDS: i <> j} x[i,j] = 1;

# Contraintes MTZ pour éliminer les sous-tours
# Si on va de i à j (i,j clients), alors u[i] + 1 <= u[j] + n*(1-x[i,j])
s.t. elimination_sous_tours{i in CLIENTS, j in CLIENTS: i <> j}:
    u[i] - u[j] + card(CLIENTS) * x[i,j] <= card(CLIENTS) - 1;

# Pas de boucle sur soi-même
s.t. pas_de_boucle{i in NOEUDS}:
    x[i,i] = 0;

end;
